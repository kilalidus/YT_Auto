   101→function getInitials(name: string): string {
   102→  if (!name) return 'U'
   103→  const parts = name.trim().split(/\s+/)
   104→  if (parts.length === 1) return parts[0].slice(0, 2).toUpperCase()
   105→  return (parts[0][0] + parts[parts.length - 1][0]).toUpperCase()
   106→}

   127→function NotifToggle({
   128→  id,
   129→  label,
   130→  description,
   131→  icon: Icon,
   132→  color,
   133→  value,
   134→  onChange,
   135→}: {
   136→  id: string
   137→  label: string
   138→  description: string
   139→  icon: typeof Bell
   140→  color: string
   141→  value: boolean
   142→  onChange: (v: boolean) => void
   143→}) {
   144→  const [pending, setPending] = useState(false)
   145→  const handle = async (v: boolean) => {
   146→    setPending(true)
   147→    try {
   148→      await apiFetch('/api/settings', {
   149→        method: 'PATCH',
   150→        body: JSON.stringify({ [id]: v }),
   151→      })
   152→      onChange(v)
   153→      toast.success(`${label} ${v ? 'enabled' : 'disabled'}`)
   154→    } catch {
   155→      toast.error(`Failed to update ${label.toLowerCase()}`)
   156→    } finally {
   157→      setPending(false)
   158→    }
   159→  }
   160→  return (
   161→    <div className="glass rounded-2xl p-4 flex items-center justify-between gap-3 lift">
   162→      <div className="flex items-center gap-3 min-w-0">
   163→        <div
   164→          className={`w-10 h-10 rounded-xl ${color} flex items-center justify-center shrink-0`}
   165→        >
   166→          <Icon className="w-4 h-4 text-white" />
   167→        </div>
   168→        <div className="min-w-0">
   169→          <Label className="text-sm font-medium cursor-pointer" htmlFor={id}>
   170→            {label}
   171→          </Label>
   172→          <p className="text-xs text-muted-foreground truncate">{description}</p>
   173→        </div>
   174→      </div>
   175→      <Switch id={id} checked={value} disabled={pending} onCheckedChange={handle} />
   176→    </div>
   177→  )
   178→}
