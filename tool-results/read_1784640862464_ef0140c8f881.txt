   336→function ContentIdeasTab({ onGoToWorkflow }: { onGoToWorkflow: () => void }) {
   337→  const [channels, setChannels] = useState<Channel[]>([])
   338→  const [selectedChannelId, setSelectedChannelId] = useState<string>('')
   339→  const [loadingChannels, setLoadingChannels] = useState(true)
   340→
   341→  const [form, setForm] = useState({
   342→    niche: '',
   343→    audience: '',
   344→    channelName: '',
   345→    count: 8,
   346→  })
   347→  const [generating, setGenerating] = useState(false)
   348→  const [ideas, setIdeas] = useState<ContentIdea[]>([])
   349→  const [sentIds, setSentIds] = useState<Set<string>>(new Set())
   350→  const [sendingId, setSendingId] = useState<string | null>(null)
   351→
   352→  // Load channels and prefill defaults
   353→  useEffect(() => {
   354→    let cancelled = false
   355→    apiFetch<{ channels: Channel[] }>('/api/channels')
   356→      .then((r) => {
   357→        if (cancelled) return
   358→        setChannels(r.channels)
   359→        if (r.channels[0]) {
   360→          const c = r.channels[0]
   361→          setSelectedChannelId(c.id)
   362→          setForm((f) => ({
   363→            ...f,
   364→            channelName: c.title,
   365→            niche: deriveNiche(c),
   366→            audience: '',
   367→          }))
   368→        }
   369→      })
   370→      .catch(() => toast.error('Failed to load channels'))
   371→      .finally(() => !cancelled && setLoadingChannels(false))
   372→    return () => {
   373→      cancelled = true
   374→    }
   375→  }, [])
   376→
   377→  const onChannelChange = (id: string) => {
   378→    setSelectedChannelId(id)
   379→    const c = channels.find((c) => c.id === id)
   380→    if (c) {
   381→      setForm((f) => ({
   382→        ...f,
   383→        channelName: c.title,
   384→        niche: f.niche || deriveNiche(c),
   385→      }))
   386→    }
   387→  }
   388→
   389→  const generate = async () => {
   390→    if (!form.niche.trim()) {
   391→      toast.error('Enter a niche or topic to generate ideas')
   392→      return
   393→    }
   394→    if (!form.audience.trim()) {
   395→      toast.error('Enter a target audience')
   396→      return
   397→    }
   398→    setGenerating(true)
   399→    setIdeas([])
   400→    setSentIds(new Set())
   401→    try {
   402→      const res = await apiFetch<{ ideas: ContentIdea[] }>('/api/ai/ideas', {
   403→        method: 'POST',
   404→        body: JSON.stringify({
   405→          niche: form.niche.trim(),
   406→          audience: form.audience.trim(),
   407→          channelName: form.channelName.trim() || 'My Channel',
   408→          count: form.count,
   409→        }),
   410→      })
   411→      setIdeas(res.ideas || [])
   412→      toast.success(`${res.ideas?.length ?? 0} ideas generated`)
   413→    } catch (e) {
   414→      toast.error(e instanceof Error ? e.message : 'Generation failed')
   415→    } finally {
   416→      setGenerating(false)
   417→    }
   418→  }
   419→
   420→  const sendToWorkflow = async (idea: ContentIdea, index: number) => {
   421→    const id = `${idea.title}-${index}`
   422→    if (sentIds.has(id)) return
   423→    setSendingId(id)
   424→    try {
   425→      const labels = [...(idea.tags ?? []), idea.format].filter(Boolean)
   426→      await apiFetch('/api/tasks', {
   427→        method: 'POST',
   428→        body: JSON.stringify({
   429→          title: idea.title,
   430→          description: `${idea.hook}\n\nWhy it works: ${idea.why}`,
   431→          status: 'idea',
   432→          labels,
   433→        }),
   434→      })
   435→      setSentIds((s) => new Set(s).add(id))
   436→      toast.success('Added to Ideas column', {
   437→        description: 'Open the workflow board to plan it.',
   438→        action: {
   439→          label: 'View board',
   440→          onClick: onGoToWorkflow,
   441→        },
   442→      })
   443→    } catch (e) {
   444→      toast.error(e instanceof Error ? e.message : 'Failed to send to workflow')
   445→    } finally {
   446→      setSendingId(null)
   447→    }
   448→  }
   449→
   450→  const copyIdea = async (idea: ContentIdea) => {
   451→    const text = `${idea.title}\n\nHook: ${idea.hook}\nFormat: ${idea.format}\nDifficulty: ${idea.difficulty}\nEstimated views: ${idea.estimatedViews}\nTags: ${idea.tags.join(', ')}`
   452→    try {
   453→      await navigator.clipboard.writeText(text)
   454→      toast.success('Idea copied to clipboard')
   455→    } catch {
   456→      toast.error('Failed to copy')
   457→    }
   458→  }
   459→
   460→  const hasChannels = channels.length > 0
   461→
   462→  return (
   463→    <div className="grid lg:grid-cols-[340px_1fr] gap-4">
   464→      {/* Left form panel */}
   465→      <motion.div
   466→        initial={{ opacity: 0, x: -10 }}
   467→        animate={{ opacity: 1, x: 0 }}
   468→        transition={{ delay: 0.05 }}
   469→      >
   470→        <Card className="glass border-border/60 lg:sticky lg:top-4">
   471→          <CardHeader className="pb-3">
   472→            <CardTitle className="text-base flex items-center gap-2">
   473→              <Wand2 className="w-4 h-4 text-primary" /> Brainstorm Setup
   474→            </CardTitle>
   475→            <CardDescription className="text-xs">
   476→              Tell Gemini your niche & audience — get fresh, varied video
   477→              concepts.
   478→            </CardDescription>
   479→          </CardHeader>
   480→          <CardContent className="space-y-3.5">
   481→            <div className="space-y-1.5">
   482→              <Label>Channel</Label>
   483→              {loadingChannels ? (
   484→                <div className="h-9 rounded-xl glass shimmer" />
   485→              ) : hasChannels ? (
   486→                <Select
   487→                  value={selectedChannelId}
   488→                  onValueChange={onChannelChange}
   489→                >
   490→                  <SelectTrigger className="w-full rounded-xl">
   491→                    <SelectValue placeholder="Select channel" />
   492→                  </SelectTrigger>
   493→                  <SelectContent>
   494→                    {channels.map((c) => (
   495→                      <SelectItem key={c.id} value={c.id}>
   496→                        {c.title}
   497→                      </SelectItem>
   498→                    ))}
   499→                  </SelectContent>
   500→                </Select>
   501→              ) : (
   502→                <p className="text-[11px] text-muted-foreground italic">
   503→                  No channels yet — you can still type the fields below
   504→                  manually.
   505→                </p>
   506→              )}
   507→            </div>
   508→
   509→            <div className="space-y-1.5">
   510→              <Label htmlFor="niche">
   511→                Niche / topic <span className="text-red-400">*</span>
   512→              </Label>
   513→              <Input
   514→                id="niche"
   515→                placeholder="e.g. Personal finance for beginners"
   516→                value={form.niche}
   517→                onChange={(e) => setForm({ ...form, niche: e.target.value })}
   518→                className="rounded-xl"
   519→              />
   520→            </div>
   521→
   522→            <div className="space-y-1.5">
   523→              <Label htmlFor="audience">
   524→                Target audience <span className="text-red-400">*</span>
   525→              </Label>
   526→              <Input
   527→                id="audience"
   528→                placeholder="e.g. Beginner content creators"
   529→                value={form.audience}
   530→                onChange={(e) =>
   531→                  setForm({ ...form, audience: e.target.value })
   532→                }
   533→                className="rounded-xl"
   534→              />
   535→            </div>
   536→
   537→            <div className="space-y-1.5">
   538→              <Label htmlFor="channel-name">Channel name</Label>
   539→              <Input
   540→                id="channel-name"
   541→                placeholder="My Channel"
   542→                value={form.channelName}
   543→                onChange={(e) =>
   544→                  setForm({ ...form, channelName: e.target.value })
   545→                }
   546→                className="rounded-xl"
   547→              />
   548→            </div>
   549→
   550→            <div className="space-y-1.5">
   551→              <Label>Number of ideas</Label>
   552→              <div className="grid grid-cols-4 gap-2">
   553→                {COUNT_OPTIONS.map((n) => (
   554→                  <button
   555→                    key={n}
   556→                    type="button"
   557→                    onClick={() => setForm({ ...form, count: n })}
   558→                    className={`h-9 rounded-xl text-sm font-semibold transition-all ${
   559→                      form.count === n
   560→                        ? 'grad-primary text-primary-foreground glow-primary'
   561→                        : 'glass text-muted-foreground hover:text-foreground hover:bg-accent/40'
   562→                    }`}
   563→                  >
   564→                    {n}
   565→                  </button>
   566→                ))}
   567→              </div>
   568→            </div>
   569→
   570→            <Button
   571→              onClick={generate}
   572→              disabled={generating || !form.niche.trim() || !form.audience.trim()}
   573→              className="w-full rounded-xl grad-primary text-white glow-primary h-11"
   574→            >
   575→              {generating ? (
   576→                <>
   577→                  <motion.span
   578→                    animate={{ rotate: 360 }}
   579→                    transition={{
   580→                      duration: 1.5,
   581→                      repeat: Infinity,
   582→                      ease: 'linear',
   583→                    }}
   584→                    className="inline-flex"
   585→                  >
   586→                    <Wand2 className="w-4 h-4 mr-2" />
   587→                  </motion.span>
   588→                  Gemini is brainstorming ideas…
   589→                </>
   590→              ) : (
   591→                <>
   592→                  <Sparkles className="w-4 h-4 mr-2" />
   593→                  {ideas.length > 0 ? 'Generate Again' : 'Generate Ideas'}
   594→                </>
   595→              )}
   596→            </Button>
   597→
   598→            <p className="text-[11px] text-muted-foreground text-center">
   599→              Uses Gemini LLM · typically 10–25 seconds.
   600→            </p>
   601→          </CardContent>
   602→        </Card>
   603→      </motion.div>
   604→
   605→      {/* Right results panel */}
   606→      <motion.div
   607→        initial={{ opacity: 0, x: 10 }}
   608→        animate={{ opacity: 1, x: 0 }}
   609→        transition={{ delay: 0.1 }}
   610→        className="space-y-4"
   611→      >
   612→        {generating ? (
   613→          <IdeasLoadingSkeleton count={Math.min(form.count, 6)} />
   614→        ) : ideas.length === 0 ? (
   615→          <IdeasEmptyState />
   616→        ) : (
   617→          <>
   618→            <div className="flex items-center justify-between flex-wrap gap-2">
   619→              <div className="flex items-center gap-2 text-xs text-muted-foreground">
   620→                <Badge variant="outline" className="border-primary/30 text-primary">
   621→                  <Sparkles className="w-3 h-3 mr-1" />
   622→                  {ideas.length} ideas
   623→                </Badge>
   624→                <span>· Click an idea to send it to your workflow board.</span>
   625→              </div>
   626→              <Button
   627→                size="sm"
   628→                variant="outline"
   629→                className="rounded-xl"
   630→                onClick={generate}
   631→                disabled={generating}
   632→              >
   633→                <RotateCcw className="w-3.5 h-3.5 mr-1" /> Regenerate
   634→              </Button>
   635→            </div>
   636→
   637→            <div className="grid grid-cols-1 xl:grid-cols-2 gap-4">
   638→              {ideas.map((idea, i) => (
   639→                <IdeaCard
   640→                  key={`${idea.title}-${i}`}
   641→                  idea={idea}
   642→                  index={i}
   643→                  sent={sentIds.has(`${idea.title}-${i}`)}
   644→                  sending={sendingId === `${idea.title}-${i}`}
   645→                  onSend={() => sendToWorkflow(idea, i)}
   646→                  onCopy={() => copyIdea(idea)}
   647→                />
