   103→function initials(name: string) {
   104→  return name
   105→    .split(' ')
   106→    .map((p) => p[0])
   107→    .filter(Boolean)
   108→    .slice(0, 2)
   109→    .join('')
   110→    .toUpperCase() || 'U'
   111→}

  1023→function NotifToggle({
  1024→  id,
  1025→  label,
  1026→  description,
  1027→  icon: Icon,
  1028→  color,
  1029→  value,
  1030→  onSaved,
  1031→  patch,
  1032→}: {
  1033→  id: string
  1034→  label: string
  1035→  description: string
  1036→  icon: typeof Bell
  1037→  color: string
  1038→  value: boolean
  1039→  onSaved: (v: boolean) => void
  1040→  patch: (p: Partial<UserSettings>) => Promise<void>
  1041→}) {
  1042→  const [pending, setPending] = useState(false)
  1043→  const [local, setLocal] = useState(value)
  1044→
  1045→  // keep in sync if upstream changes
  1046→  useEffect(() => {
  1047→    setLocal(value)
  1048→  }, [value])
  1049→
  1050→  const handle = async (v: boolean) => {
  1051→    setLocal(v)
  1052→    setPending(true)
  1053→    try {
  1054→      await patch({ [id]: v })
  1055→      onSaved(v)
  1056→      toast.success(`${label} ${v ? 'enabled' : 'disabled'}`)
  1057→    } catch {
  1058→      setLocal(!v)
  1059→      // toast already shown by patch wrapper
  1060→    } finally {
  1061→      setPending(false)
  1062→    }
  1063→  }
  1064→
  1065→  return (
  1066→    <div className="rounded-2xl border border-border/60 p-4 bg-background/30 flex items-start gap-3">
  1067→      <div className={`w-9 h-9 rounded-lg ${color} flex items-center justify-center shrink-0`}>
  1068→        <Icon className="w-4 h-4 text-white" />
  1069→      </div>
  1070→      <div className="flex-1 min-w-0">
  1071→        <Label className="text-sm font-medium block">{label}</Label>
  1072→        <p className="text-xs text-muted-foreground mt-0.5">{description}</p>
  1073→      </div>
  1074→      <Switch checked={local} disabled={pending} onCheckedChange={handle} />
  1075→    </div>
  1076→  )
  1077→}
