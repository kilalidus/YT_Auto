   101тЖТ   101тЖТfunction getInitials(name: string): string {
   102тЖТ   102тЖТ  if (!name) return 'U'
   103тЖТ   103тЖТ  const parts = name.trim().split(/\s+/)
   104тЖТ   104тЖТ  if (parts.length === 1) return parts[0].slice(0, 2).toUpperCase()
   105тЖТ   105тЖТ  return (parts[0][0] + parts[parts.length - 1][0]).toUpperCase()
   106тЖТ   106тЖТ}
